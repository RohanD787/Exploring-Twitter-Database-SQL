CREATE TABLE users(user_id BIGINT PRIMARY KEY,
screen_name VARCHAR(255) NOT NULL UNIQUE,
location VARCHAR(255),
utc_offset INT,
time_zone VARCHAR(127),
follower_count INT,
language VARCHAR(10),
description VARCHAR(255),
status_count VARCHAR(255),
friends_count INT,
created_at TIMESTAMP WITH TIME ZONE NOT NULL);


CREATE TABLE Tweet (
tweet_id BIGINT PRIMARY KEY,
created_at TIMESTAMP WITH TIME ZONE,
text TEXT,
in_reply_to_user_id BIGINT,
retweet_count INT,
source VARCHAR(255),
retweet_tweet_id BIGINT,
user_id BIGINT NOT NULL,
FOREIGN KEY (user_id) REFERENCES users(user_id),
FOREIGN KEY (in_reply_to_user_id) REFERENCES users(user_id),
FOREIGN KEY (retweet_tweet_id) REFERENCES tweet(tweet_id) );

/* In table Tweet, 'in_reply_to_user_id' may not be present in the table 'users', it may give an error while referencing. */


CREATE TABLE Hashtag(
hashtag_id SERIAL PRIMARY KEY,
hashtag1 varchar(144),
hashtag2 varchar(144),
hashtag3 varchar(144),
hashtag4 varchar(144),
hashtag5 varchar(144),
hashtag6 varchar(144) );

CREATE TABLE Tweet_Hashtag(
tweet_id BIGINT,
hashtag_id BIGINT,
FOREIGN KEY (tweet_id) REFERENCES Tweet(tweet_id),
FOREIGN KEY (hashtag_id) REFERENCES Hashtag(hashtag_id) );
